@startuml
skinparam classAttributeIconSize 0

class Tuile {
  + id : std::string
  + minPlayers : int
}

class Tas {
  - nb_tuiles : size_t
  - tuiles : std::vector<const Tuile*>
  --
  + Tas(nbjoueurs : int)
  + getTuilesTas(i : size_t) : const Tuile*
  + getNbTuiles() : size_t
  + pushTuile(t : const Tuile*) : void
  ~ Tas()
}

class Pioche {
  - tas : std::vector<const Tas*>
  - curseur : std::vector<size_t>
  --
  + ajouterTas(t : const Tas*) : void
  + estVide() : bool
  + piocherDepuisTas(i : size_t) : const Tuile*
  + prendreIndiceTasNonVide() : int
  ~ Pioche()
}

class Table {
  - capacite : size_t
  - visibles : std::vector<const Tuile*>
  - reservoir : std::vector<const Tuile*>
  --
  + Table(nb_joueurs : size_t, res : std::vector<const Tuile*>)
  + taille() : size_t
  + estVide() : bool
  + cout(i : size_t) : size_t
  + print() : void
  + prendre(i : size_t) : const Tuile*
  + recharger(p : Pioche&) : void
  --
  - remplirDepuisReservoir() : void
  - remplirDepuisReservoirOuTas(p : Pioche&) : void
  - regleUneTuileRestante(p : Pioche&) : void
}

' ========= Relations =========
Tas "1" o-- "*" Tuile : tuiles
Pioche "1" o-- "11" Tas : tas
Table "1" o-- "0..*" Tuile : visibles / reservoir
Table ..> Pioche : recharger()

@enduml
